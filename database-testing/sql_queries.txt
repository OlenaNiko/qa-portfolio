create Database courseDB;

CREATE TABLE courseDB.Universities (
Id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
UniName varchar (50) NOT NULL,
City varchar (50) NOT NULL
);

CREATE TABLE courseDB.Faculties (
Id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
Name varchar (50) NOT NULL,
UniversityId INT,
FOREIGN KEY (UniversityId) REFERENCES Universities(Id)
);

CREATE TABLE courseDB.SGroups (
Id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
GroupNumber INT NOT NULL,
FacultyId INT,
FOREIGN KEY (FacultyId) REFERENCES Faculties(Id)
);

CREATE TABLE courseDB.Students (
Id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
Name varchar (50) NOT NULL,
Surname varchar (50) NOT NULL,
Age INT,
GroupId INT,
FOREIGN KEY (GroupId) REFERENCES SGroups(Id)
);
